package com.example.europroject.models;

import java.util.List;

public abstract class CollectionService {

    public final static class CollectionTransferObject {
        public final int mFirst;
        public final int mLength;
        public final int mTotal;
        public final List<Collection> mData;

        public CollectionTransferObject(final List<Collection> data, final int first,
                                  final int length, final int total) {
            this.mData = data;
            this.mFirst = first;
            this.mLength = length;
            this.mTotal = total;
        }
    }

    public abstract CollectionTransferObject loadCollections(final int start, final int length);
}
